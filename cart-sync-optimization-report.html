async function ensureCartAssociation() { try { // 檢查購物車狀態 const cart = await getCurrentCart() if (!cart.customer_id) { // 重新嘗試轉移 await transferCart() } } catch (error) { // 轉移失敗時清除本地購物車 await removeCartId() // 系統會自動創建新的關聯購物車 } }
